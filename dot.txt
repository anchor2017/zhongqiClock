/****************************************************************************
*函数名：GUI_Dot
*输  入：x：点的X坐标
*      * y：点的Y坐标
*      * color：点的颜色
*输  出：
*功  能：给一块3*3像素涂上颜色。
****************************************************************************/	  

void GUI_Dot(uint x, uint y, uint color)
{  
	uchar i;

	TFT_SetWindow(x - 1, y - 1, x + 2, y + 2);  //单个像素

	for(i=0; i<16; i++)
	{
		TFT_WriteColor(color);
	}
}
